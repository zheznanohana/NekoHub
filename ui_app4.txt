# Optional storage: use your existing storage.py if present
try:
    from storage import init_db, upsert_messages, latest_messages, get_meta, set_meta  # type: ignore
except Exception:
    init_db = lambda: None  # type: ignore
    _MEM_DB: List[Tuple[int, int, str, str, int, str]] = []
    _META = {"last_id": "0"}

    def upsert_messages(msgs: List[dict]):  # type: ignore
        for m in msgs:
            mid = int(m.get("id", 0))
            appid = int(m.get("appid", 0))
            title = str(m.get("title") or "Gotify")
            msg = str(m.get("message") or "")
            prio = int(m.get("priority", 0))
            date = str(m.get("date") or "")
            if any(r[0] == mid for r in _MEM_DB):
                continue
            _MEM_DB.append((mid, appid, title, msg, prio, date))
        _MEM_DB.sort(key=lambda x: x[0], reverse=True)

    def latest_messages(limit: int):  # type: ignore
        return _MEM_DB[: max(0, int(limit))]

    def get_meta(k: str):  # type: ignore
        return _META.get(k)

    def set_meta(k: str, v: str):  # type: ignore
        _META[k] = v

class UiRefreshCoalescer:
    def __init__(self, fn, interval_ms: int = 200):
        self._fn = fn
        self._timer = QTimer()
        self._timer.setSingleShot(True)
        self._timer.timeout.connect(self._fn)
        self._interval = max(50, int(interval_ms))

    def request(self):
        if not self._timer.isActive():
            self._timer.start(self._interval)


class TrayController:
    def __init__(self, window: QWidget, title: str = "NekoHub", icon: Optional[QIcon] = None):
        self.window = window
        self.tray = QSystemTrayIcon(window)
        self.tray.setToolTip(title)

        if icon and not icon.isNull():
            self.tray.setIcon(icon)
        else:
            self.tray.setIcon(QApplication.style().standardIcon(QStyle.SP_ComputerIcon))

        menu = QMenu()
        act_show = menu.addAction("Show / Restore")
        act_exit = menu.addAction("Exit")
        act_show.triggered.connect(self.restore)
        act_exit.triggered.connect(self.exit_app)
        self.tray.setContextMenu(menu)

        self.tray.activated.connect(self._on_tray_activated)
        self.tray.show()

    def _on_tray_activated(self, reason):
        if reason in (QSystemTrayIcon.Trigger, QSystemTrayIcon.DoubleClick):
            self.restore()

    def restore(self):
        self.window.showNormal()
        self.window.raise_()
        self.window.activateWindow()

    def exit_app(self):
        QApplication.quit()


class InboxPage(QWidget):
    def __init__(self):
        super().__init__()
        self.setObjectName("inbox")
        # 性能优化：限制列表渲染
        self._limit = 100  
        self._unread = 0

        root = QVBoxLayout(self)
        root.setContentsMargins(24, 18, 24, 18)
        root.setSpacing(10)

        top = QHBoxLayout()
        self.lbl_title = TitleLabel("Inbox")
        top.addWidget(self.lbl_title, 0)

        self.lbl_unread = BodyLabel("")
        self.lbl_unread.setStyleSheet("color:#ff3b30; font-weight:700; padding-left:6px;")
        self.lbl_unread.hide()
        top.addWidget(self.lbl_unread, 0)

        top.addStretch(1)

        self.btn_mark = PushButton("Mark read")
        self.btn_mark.clicked.connect(self.clear_unread)
        top.addWidget(self.btn_mark, 0)
        root.addLayout(top)

        self.lbl_status = BodyLabel("WS: idle")
        self.lbl_status.setStyleSheet("opacity:0.75;")
        root.addWidget(self.lbl_status)

        self.lbl_hint = BodyLabel("Waiting for messages…")
        self.lbl_hint.setStyleSheet("opacity:0.75;")
        root.addWidget(self.lbl_hint)

        self.scroll = QScrollArea()
        self.scroll.setWidgetResizable(True)
        self.scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarAlwaysOff)
        self.scroll.setVerticalScrollBarPolicy(Qt.ScrollBarAsNeeded)

        self.container = QWidget()
        self.v = QVBoxLayout(self.container)
        self.v.setSpacing(10)
        self.v.setContentsMargins(0, 0, 0, 0)
        self.scroll.setWidget(self.container)
        root.addWidget(self.scroll, 1)

        self._cards: List[CardWidget] = []

    def set_status(self, s: str):
        self.lbl_status.setText(s)

    def set_hint(self, s: str):
        self.lbl_hint.setText(s)

    def limit(self) -> int:
        return self._limit

    def add_unread(self, n: int = 1):
        self._unread = max(0, self._unread + int(n))
        self._sync_unread()

    def clear_unread(self):
        self._unread = 0
        self._sync_unread()

    def _sync_unread(self):
        if self._unread <= 0:
            self.lbl_unread.hide()
        else:
            self.lbl_unread.setText(f"● {self._unread}")
            self.lbl_unread.show()

    def _make_card(self, row: Tuple) -> CardWidget:
        mid, appid, title, msg, prio, date = row
        card = CardWidget()
        card.setBorderRadius(12)
        lay = QVBoxLayout(card)
        lay.setContentsMargins(14, 12, 14, 12)
        lay.setSpacing(6)

        head = QHBoxLayout()
        head.addWidget(SubtitleLabel(str(title) if title else "Gotify"), 0)
        head.addStretch(1)
        meta = BodyLabel(f"#{mid}  app:{appid}  p:{prio}")
        meta.setStyleSheet("opacity:0.65;")
        head.addWidget(meta, 0)
        lay.addLayout(head)

        body = BodyLabel(str(msg) if msg else "")
        body.setWordWrap(True)
        lay.addWidget(body)

        if date:
            dt = BodyLabel(str(date))
            dt.setStyleSheet("opacity:0.5;")
            lay.addWidget(dt)

        return card

    def refresh_full(self, rows: List[Tuple]):
        for c in self._cards:
            c.setParent(None)
        self._cards.clear()

        if not rows:
            self.set_hint("No messages yet.")
            return

        self.set_hint(f"Showing {len(rows)} messages.")
        for r in rows:
            card = self._make_card(r)
            self.v.addWidget(card)
            self._cards.append(card)

        self.v.addStretch(1)

    def insert_top(self, row: Tuple):
        card = self._make_card(row)
        self.v.insertWidget(0, card)
        self._cards.insert(0, card)

        while len(self._cards) > self._limit:
            last = self._cards.pop()
            last.setParent(None)


class SettingsPage(QWidget):
    def __init__(self, get_cb, save_cb, test_send_cb, reconnect_cb):
        super().__init__()
        self.setObjectName("settings")
        self._get = get_cb
        self._save = save_cb
        self._test_send = test_send_cb
        self._reconnect = reconnect_cb

        root = QVBoxLayout(self)
        root.setContentsMargins(24, 18, 24, 18)
        root.setSpacing(12)

        root.addWidget(TitleLabel("Settings"))

        card = CardWidget()
        card.setBorderRadius(12)
        lay = QVBoxLayout(card)
        lay.setContentsMargins(16, 14, 16, 14)
        lay.setSpacing(12)

        lay.addWidget(SubtitleLabel("Gotify"))

        self.ed_url = LineEdit()
        self.ed_url.setPlaceholderText("Gotify URL")

        self.ed_recv = LineEdit()
        self.ed_recv.setPlaceholderText("Receive Token (Client token, listen/get)")
        self.ed_recv.setEchoMode(QLineEdit.Password)

        self.ed_send = LineEdit()
        self.ed_send.setPlaceholderText("Send Token (App token, post)")
        self.ed_send.setEchoMode(QLineEdit.Password)

        self.sp_poll = QSpinBox()
        self.sp_poll.setRange(2, 60)
        self.sp_poll.setSuffix(" s")

        lay.addWidget(SubtitleLabel("Notification"))
        self.sw_sound = SwitchButton("Enable sound (Windows default)")
        self.sw_sound.setChecked(True)

        def row(label: str, w: QWidget):
            r = QHBoxLayout()
            r.addWidget(BodyLabel(label), 0)
            r.addWidget(w, 1)
            return r

        lay.addLayout(row("URL", self.ed_url))
        lay.addLayout(row("Receive", self.ed_recv))
        lay.addLayout(row("Send", self.ed_send))

        r_poll = QHBoxLayout()
        r_poll.addWidget(BodyLabel("Poll seconds (history fetch)"), 0)
        r_poll.addWidget(self.sp_poll, 0)
        r_poll.addStretch(1)
        lay.addLayout(r_poll)

        lay.addWidget(self.sw_sound)

        btns = QHBoxLayout()
        self.btn_save = PushButton("Save")
        self.btn_test = PushButton("Send Test")
        self.btn_reconn = PushButton("Reconnect WS")
        btns.addWidget(self.btn_save)
        btns.addWidget(self.btn_test)
        btns.addWidget(self.btn_reconn)
        btns.addStretch(1)

        self.btn_save.clicked.connect(self._on_save)
        self.btn_test.clicked.connect(self._on_test)
        self.btn_reconn.clicked.connect(self._on_reconnect)

        lay.addLayout(btns)

        root.addWidget(card)
        root.addStretch(1)

        self.load_to_ui()

    def load_to_ui(self):
        s = self._get()
        self.ed_url.setText(s.gotify_url)
        self.ed_recv.setText(s.gotify_recv_token)
        self.ed_send.setText(s.gotify_send_token)
        self.sp_poll.setValue(int(getattr(s, "poll_seconds", 2)))
        self.sw_sound.setChecked(bool(getattr(s, "sound_enabled", True)))

    def _collect(self) -> Settings:
        s = self._get()
        d = asdict(s)
        d["gotify_url"] = self.ed_url.text().strip()
        d["gotify_recv_token"] = self.ed_recv.text().strip()
        d["gotify_send_token"] = self.ed_send.text().strip()
        d["poll_seconds"] = int(self.sp_poll.value())
        d["sound_enabled"] = bool(self.sw_sound.isChecked())
        return Settings(**d)

    def _on_save(self):
        s = self._collect()
        self._save(s)
        QMessageBox.information(self, "Saved", "Settings saved.")

    def _on_test(self):
        ok, err = self._test_send(self._collect())
        if ok:
            QMessageBox.information(self, "OK", "Test message sent.")
        else:
            QMessageBox.critical(self, "Fail", err or "Unknown error")

    def _on_reconnect(self):
        self._reconnect()
        QMessageBox.information(self, "OK", "WebSocket reconnect triggered.")


class UiBridge(QObject):
    status = Signal(str)
    message = Signal(dict)
    refresh_request = Signal()


# 性能优化：初始拉取数量
INITIAL_HISTORY_LIMIT = 50


class NekoHubWindow(FluentWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("NekoHub")
        self.resize(1100, 720)
        self.setMinimumSize(980, 620)

        icon = QIcon("icon.ico")
        if icon.isNull():
            icon = QApplication.style().standardIcon(QStyle.SP_ComputerIcon)
        self.setWindowIcon(icon)

        init_db()
        self._settings: Settings = load_settings()
        self._last_id = int(get_meta("last_id") or "0")

        self.inbox = InboxPage()
        self.settings_page = SettingsPage(self.get_settings, self.save_settings, self.send_test, self.reconnect_ws)
        
        # 初始化 AI 模块
        self.ai_manager = AiManager(self._settings)
        self.ai_page = AiPage(self.ai_manager, self.get_settings, self.save_settings)

        self.addSubInterface(self.inbox, FluentIcon.MAIL, "Inbox")
        self.addSubInterface(self.ai_page, FluentIcon.CHAT, "AI Chat")
        self.addSubInterface(self.settings_page, FluentIcon.SETTING, "Settings", position=NavigationItemPosition.BOTTOM)

        self.tray = TrayController(self, title="NekoHub", icon=icon)

        self.sound = SoundPlayer()
        self.toast_q = ToastQueue(max_visible=6, life_ms=3200)

        self._refresh_coalescer = UiRefreshCoalescer(self._refresh_full_once, interval_ms=200)

        self._bridge = UiBridge()
        self._bridge.status.connect(self._ui_set_status)
        self._bridge.message.connect(self._ui_handle_message)
        self._bridge.refresh_request.connect(self._ui_refresh_request)

        self._soft_timer = QTimer(self)
        self._soft_timer.timeout.connect(self._soft_tick)
        self._soft_timer.start(6000)

        self._ws = None

        self._refresh_full_once()
        self._initial_history_fetch_async()
        self._start_ws()

    def closeEvent(self, e):
        e.ignore()
        self.hide()

    def get_settings(self) -> Settings:
        return self._settings

    def save_settings(self, s: Settings):
        self._settings = s
        save_settings(s)
        self.sound.enabled = bool(getattr(s, "sound_enabled", True))
        
        if hasattr(self, 'ai_manager'):
            self.ai_manager.update_settings(s)
            
        self._initial_history_fetch_async()
        self._start_ws()
        self._refresh_coalescer.request()

    def reconnect_ws(self):
        self._start_ws()

    def send_test(self, s: Settings):
        try:
            if not s.gotify_url or not s.gotify_send_token:
                return False, "Missing Gotify URL or Send Token (App token)"
            if send_message is None:
                return False, "gotify_client.send_message not found"
            send_message(s.gotify_url, s.gotify_send_token, title="NekoHub Test", message="Test push OK", priority=5)
            return True, ""
        except Exception as e:
            return False, str(e)

    def _soft_tick(self):
        self._refresh_coalescer.request()
        try:
            cur = getattr(self, "stackedWidget", None)
            if cur and cur.currentWidget() == self.inbox:
                self.inbox.clear_unread()
        except Exception:
            pass

    def _refresh_full_once(self):
        rows = latest_messages(self.inbox.limit())
        self.inbox.refresh_full(rows)

    @Slot()
    def _ui_refresh_request(self):
        self._refresh_coalescer.request()

    @Slot(str)
    def _ui_set_status(self, s: str):
        self.inbox.set_status(s)

    @Slot(dict)
    def _ui_handle_message(self, m: dict):
        try:
            upsert_messages([m])
        except Exception:
            pass

        try:
            mid = int(m.get("id", 0))
        except Exception:
            mid = 0

        is_new = bool(mid and mid > self._last_id)
        if is_new:
            self._last_id = mid
            set_meta("last_id", str(self._last_id))

        title = str(m.get("title") or "Gotify")
        body = str(m.get("message") or "")

        if is_new:
            self.sound.play()
            self.toast_q.show(title, body)

            self.inbox.add_unread(1)
            
            # 通知 AI 进行计数
            self.ai_manager.on_new_message(m)

            rows = latest_messages(1)
            if rows:
                self.inbox.insert_top(rows[0])

    def _initial_history_fetch_async(self):
        def _work():
            try:
                s = self._settings
                if not s.gotify_url or not s.gotify_recv_token:
                    return
                if fetch_latest is None:
                    return
                msgs = fetch_latest(s.gotify_url, s.gotify_recv_token, limit=INITIAL_HISTORY_LIMIT)
                if msgs:
                    upsert_messages(msgs)
                    mx = 0
                    for mm in msgs:
                        try:
                            mx = max(mx, int(mm.get("id", 0)))
                        except Exception:
                            pass
                    if mx > self._last_id:
                        self._last_id = mx
                        set_meta("last_id", str(self._last_id))

                self._bridge.refresh_request.emit()
            except Exception:
                pass

        threading.Thread(target=_work, daemon=True).start()

    def _start_ws(self):
        try:
            if self._ws:
                self._ws.stop()
        except Exception:
            pass
        self._ws = None

        s = self._settings
        if not s.gotify_url or not s.gotify_recv_token:
            self.inbox.set_hint("Please set Gotify URL + Receive Token in Settings.")
            self.inbox.set_status("WS: idle")
            return

        if GotifyWSReceiver is None:
            self.inbox.set_status("WS: websocket-client missing (install websocket-client)")
            return

        def on_status(text: str):
            self._bridge.status.emit(text)

        def on_msg(mm: dict):
            self._bridge.message.emit(mm)

        self.inbox.set_status("WS: connecting…")

        self._ws = GotifyWSReceiver(
            base_url=s.gotify_url,
            recv_token=s.gotify_recv_token,
            on_message=on_msg,
            on_status=on_status,
            reconnect_delay=3.0,
        )
        self._ws.start()

    app = QApplication(sys.argv)
